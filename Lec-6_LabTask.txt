3.
select empno,ename, sal,round((sal+sal*1.155),0) "New Salary" ,round((sal+sal*0.155), 0)-sal "Increase"
from   emp;
----------------------------------------------------------------------------------------------------------------------------------------------------------------
4.
SELECT ename, hiredate,TO_CHAR(NEXT_DAY(ADD_MONTHS(hiredate, 6),'sunday'),'fmDay, "the" Ddspth "of" Month, YYYY') REVIEW
 FROM emp; 
----------------------------------------------------------------------------------------------------------------------------------------------------------------
5.
SELECT ename,ROUND(MONTHS_BETWEEN(SYSDATE,hiredate))MONTHS_WORKED
FROM emp
ORDER BY MONTHS_WORKED;
----------------------------------------------------------------------------------------------------------------------------------------------------------------- 
6.
SELECT ename || ' earns '|| TO_CHAR(sal, 'fm$99,999.00')|| ' monthly but wants '|| TO_CHAR(sal * 3, 'fm$99,999.00')|| '.' "Dream Salaries"
FROM emp;
------------------------------------------------------------------------------------------------------------------------------------------------------------------
7.
SELECT INITCAP(ename) "Name",LENGTH(ename) "Length"
FROM emp
WHERE ename LIKE 'J%'
OR ename LIKE 'M%'
OR ename LIKE 'A%'
ORDER BY ename ;
-------------------------------------------------------------------------------------------------------------------------------------------------------------------- 
8.
SELECT ename,NVL(TO_CHAR(comm),'No Commission') COMM
FROM emp; 
---------------------------------------------------------------------------------------------------------------------------------------------------------------------
9.
SELECT CONCAT(ename, RPAD(' ', (TRUNC(sal, -2) / 100 + 1), '*'))
          AS "EMPLOYEE_AND_THEIR_SALARIES"
FROM emp
ORDER BY sal DESC;
